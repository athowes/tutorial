library(INLA)
library(ggplot2)

n <- 100
x <- runif(n)
eta <- -1 + x
lambda <- exp(eta)
y <- rpois(n, lambda)

  family = "poisson",

2 Adding zero inflation

n <- 100
x <- runif(n)
eta <- -1 + x
lambda <- exp(eta)
p <- 0.35
z <- runif(n) > p
y <- rpois(n, lambda)
y[!z] <- 0

df <- data.frame(id = 1:n, x = x, y = y)

ggplot(df, aes(x = id, y = y)) +
  geom_point() +
  theme_minimal()

fit_zip <- INLA::inla(
  y ~ 1 + x,
  family = "zeroinflatedpoisson1",
  data = data.frame(x, y), 
  control.predictor = list(link = 1),
  control.compute=list(config = TRUE)
)

y_samples <- sapply(samples_zip, function(x) {
  predictor_sample <- x$latent[target, 1]
  lambda_sample <- exp(predictor_sample)
  stopifnot(length(lambda_sample) == n)
  y_sample <- rpois(n, lambda_sample)
  p_sample <- x$hyperpar
  z <- runif(n) > p_sample
  y[!z] <- 0
  return(y)
})
